Read DRAGONPING_CONTEXT.md. Implement Telegram alert notifications as a minimal optional feature. Off by default. Wait for confirmation after each file.

What it does
When a service goes DOWN (or recovers UP), DragonPing sends a Telegram message to a configured chat in addition to email. If Telegram is not configured via env vars the feature does nothing — no errors, no logs, completely silent. Works alongside the existing email alert system, does not replace it.

Step 2 — Telegram sender
Create app/telegram.py:
pythonimport logging
import aiohttp
from app.config import (
    TELEGRAM_ENABLED,
    TELEGRAM_BOT_TOKEN,
    TELEGRAM_CHAT_ID
)

logger = logging.getLogger(__name__)

async def send_telegram(message: str) -> bool:
    """
    Send a Telegram message.
    Returns True on success, False on any failure.
    Completely silent if ENABLE_TELEGRAM=false.
    """
    if not TELEGRAM_ENABLED:
        return False
    if not TELEGRAM_BOT_TOKEN or not TELEGRAM_CHAT_ID:
        logger.warning(
            "Telegram enabled but BOT_TOKEN or CHAT_ID missing"
        )
        return False

    url = (f"https://api.telegram.org"
           f"/bot{TELEGRAM_BOT_TOKEN}/sendMessage")
    payload = {
        "chat_id": TELEGRAM_CHAT_ID,
        "text": message,
        "parse_mode": "HTML"
    }
    try:
        async with aiohttp.ClientSession() as session:
            async with session.post(
                url, json=payload, timeout=aiohttp.ClientTimeout(total=5)
            ) as resp:
                if resp.status == 200:
                    logger.info("Telegram alert sent")
                    return True
                else:
                    body = await resp.text()
                    logger.error(
                        f"Telegram send failed {resp.status}: {body}"
                    )
                    return False
    except Exception as e:
        logger.error(f"Telegram send exception: {e}")
        return False


def format_down_message(service_name: str,
                         target: str,
                         protocol: str) -> str:
    return (
        f"🔴 <b>DragonPing Alert</b>\n\n"
        f"<b>Service DOWN</b>: {service_name}\n"
        f"<b>Target</b>: {target}\n"
        f"<b>Protocol</b>: {protocol.upper()}\n"
        f"<b>Time</b>: {__import__('datetime').datetime.now().strftime('%Y-%m-%d %H:%M:%S')}"
    )


def format_up_message(service_name: str,
                       target: str,
                       response_ms: float | None) -> str:
    resp = f"{response_ms:.0f}ms" if response_ms else "—"
    return (
        f"🟢 <b>DragonPing Alert</b>\n\n"
        f"<b>Service RECOVERED</b>: {service_name}\n"
        f"<b>Target</b>: {target}\n"
        f"<b>Response time</b>: {resp}\n"
        f"<b>Time</b>: {__import__('datetime').datetime.now().strftime('%Y-%m-%d %H:%M:%S')}"
    )


def format_cert_message(service_name: str,
                         days_left: int) -> str:
    emoji = "🔴" if days_left < 15 else "🟡"
    return (
        f"{emoji} <b>DragonPing Alert</b>\n\n"
        f"<b>SSL Certificate Expiring</b>: {service_name}\n"
        f"<b>Days remaining</b>: {days_left}\n"
        f"<b>Action</b>: Renew certificate immediately"
    )


def format_prediction_message(service_name: str,
                               reason: str) -> str:
    return (
        f"⚠️ <b>DragonPing Prediction</b>\n\n"
        f"<b>Risk detected</b>: {service_name}\n"
        f"<b>Reason</b>: {reason}\n"
        f"<b>Action</b>: Check service before it goes down"
    )
